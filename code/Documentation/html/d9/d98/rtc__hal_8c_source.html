33 void RTC_HAL_getvalue(struct tm *dateTime)
34 {
35  ctime0 = LPC_RTC->CTIME0;
36  ctime1 = LPC_RTC->CTIME1;
37  ctime2 = LPC_RTC->CTIME2;
38  dateTime->tm_sec = ctime0 & SECONDS_MASK;
39  dateTime->tm_min = (ctime0 & MINUTES_MASK)>>8;
40  dateTime->tm_hour = (ctime0 & HOURS_MASK)>>16;
41  dateTime->tm_wday = (ctime0 & WDAYS_MASK)>>24;
42  dateTime->tm_mday = (ctime1 & MDAYS_MASK);
43  dateTime->tm_mon = ((ctime1 & MONTHS_MASK)>>8) + 1;
44  dateTime->tm_year = ((ctime1 & YEARS_MASK)>>16) - 1900;
45  dateTime->tm_yday = (ctime2 & YDAYS_MASK) + 1;
46 }
